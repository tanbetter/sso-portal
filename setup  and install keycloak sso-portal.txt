Install keycloak

docker network create keycloak-network  #create network in docker
docker run --name mysql -d --net keycloak-network -p 3306:3306 -e MYSQL_DATABASE=keycloak -e MYSQL_USER=keycloak -e MYSQL_PASSWORD=password -e MYSQL_ROOT_PASSWORD=root_password mysql สร้าง mysql
docker run --name keycloak -p 8080:8080 --net keycloak-network -e KEYCLOAK_USER=admin -e KEYCLOAK_PASSWORD=admin keycloak-portal
